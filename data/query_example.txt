Linux/MacOS:
curl "http://localhost:3000/api/riot?action=account&gameName=[gameName]&tagLine=[tagLine]"
curl "http://localhost:3000/api/riot?action=match-history&puuid=[puuid]"
curl "http://localhost:3000/api/riot?action=match-info&matchId=[matchId]"
curl "http://localhost:3000/api/riot?action=match-timeline&matchId=[matchId]"

Windows:
Invoke-RestMethod -Uri "http://localhost:3000/api/riot?action=account&gameName=[gameName]&tagLine=[tagLine]"
Invoke-RestMethod -Uri "http://localhost:3000/api/riot?action=match-history&puuid=[puuid]"
Invoke-RestMethod -Uri "http://localhost:3000/api/riot?action=match-info&matchId=[matchId]"
Invoke-RestMethod -Uri "http://localhost:3000/api/riot?action=match-timeline&matchId=[matchId]"
